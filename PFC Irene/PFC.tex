\makeatletter       %Para definir \markpart como el nombre de la parte
\def\@part[#1]#2{%
   \def\partmark{#1}
   \ifnum \c@secnumdepth >-2\relax
     \refstepcounter{part}%
     \addcontentsline{toc}{part}{#1}%\thepart\hspace{1em}#1}%
   \else
     \addcontentsline{toc}{part}{#1}%
   \fi
   \markboth{}{}%
   {\thispagestyle{empty}
    \centering
    \interlinepenalty \@M
    \normalfont
    \Huge \bfseries #2\par}%
   \@endpart}
\makeatother
